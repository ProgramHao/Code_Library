#include<stdio.h>
#include<string.h>

void delete(char* a)
{
    int len=strlen(a);
    int flag[26]={0};
    int min=20;
    
    for(int i=0;i<len;i++)
    {
        flag[a[i]-'a']++;
    }
    
    for(int i=0;i<26;i++)
    {
        if(flag[i]<=min&&flag[i]!=0)
            min=flag[i];
    }
    
    for(int i=0;i<len;i++)
    {
        if(flag[a[i]-'a']>min)
        {
            printf("%c",a[i]);
        }
    }
}
int main()
{
    char arr[20]={0};
    scanf("%s",arr);
    delete(arr);
    return 0;
}
