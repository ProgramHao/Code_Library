#include <stdlib.h>
#include <stdio.h>
#include <string.h>
 
int main()
{
    int n;
     
    int a;
    while(~scanf("%d", &n))
    {
        int count[1001] = {0};
        for (int i = 0; i < n; i++)
        {
            scanf("%d", &a);
            count[a] = 1;
        }
        for (int i = 0; i < 1001; i++)
        {
            if (count[i] == 1)
            {
                printf("%d\n", i);
            }
        }
    }
     
    return 0;
}
